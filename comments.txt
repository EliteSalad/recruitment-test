        public void LoadPeopleList()
        {
            //Establish link to the database
            var connectionStringBuilder = new SqliteConnectionStringBuilder();
            using(var connection = new SqliteConnection(connectionStringBuilder.ConnectionString))
            {
                connection.Open();

                //Read data from data base
                var selectedCmd = connection.CreateCommand();
                selectedCmd.CommandText = "SELECT * FROM Employees";

                using (var reader = selectedCmd.ExecuteReader())
                {
                    while(reader.Read())                        //while iterating through database
                    {
                        var message = reader.GetString(0);      //get the first column of the database
                        Console.WriteLine(message);             //write results to console
                    }
                }
            }
        }
